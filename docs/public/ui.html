$("#tabGame").click(function(e) {
	e.preventDefault();
	$("#overlay").stop().animate({backgroundColor: "rgba(0, 0, 0, 0)"});
	$("#settings, #about").hide();
	if(!localPlayer) {
		$("#overlay").stop().animate({backgroundColor: "rgba(0, 0, 0, 0.5)"});
		$("#intro").fadeIn();
	};
	$("#tabSettings, #tabAbout").removeClass("active");
	$(this).addClass("active");
});
